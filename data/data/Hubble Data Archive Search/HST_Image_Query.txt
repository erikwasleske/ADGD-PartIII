from astroquery.mast.missions import MastMissions

missions = MastMissions(mission='hst')

# Cannot use multiple targets in astroquery query_object call
# Using the first target value of 128.00213 19.20164 in query_object call
target = '128.00213 19.20164'
radius = 'undefined'
radius_units = 'undefined'
select_cols = [
]

# Use query_object method to perform a cone search
results = missions.query_object(
    target
    , radius=radius
    , radius_units=radius_units
    , select_cols=select_cols,
    sci_obs_type='image',
    sci_aec='S',
    sci_instrume='wfc3,acs,wfpc,wfpc2')

print(results)
